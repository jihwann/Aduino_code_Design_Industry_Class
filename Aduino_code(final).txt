int ad_conv(byte channel, byte num)              // 습도 측정
{
  long sum = 0;
  byte n;

  for (n = 0; n < num; n++)
  {
    sum = sum + analogRead(channel);
  }
  return (sum / num);
}

int calc_RH10(int adval)
{
  int RH10;
  RH10 = adval + 6 * adval / 10 + 3 * adval / 100; // 1.63 * adval
  return (RH10);
}
